```python
# Integer
age = 25
print(age)

# Float
height = 5.9
print(height)

# String
name = "Ananya"
print(name)

# Boolean
is_student = True
print(is_student)
# List
fruits = ["apple", "banana", "cherry"]

# Tuple
coordinates = (10.0, 20.0)
print(coordinates)

# Dictionary
student = {"name": "Ananya", "age": 25, "grade": "A"}
print(student)

# Set
unique_numbers = {1, 2, 3, 2, 1}  # duplicates will be removed
print(unique_numbers)
```
